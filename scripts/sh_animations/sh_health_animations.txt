//Health Bonus Pulse
event HudHealthBonusPulse
{
	Animate	PlayerStatusHealthBonusImage	Alpha	"0"		Linear 0.0 0.0

	Animate	HealthIndicatorBG		BgColor		"HealthBuff"		Accel 0.0 0.0
	Animate	HealthIndicatorBG		Alpha		"200	"		Accel 0.0 0.2
	Animate	HealthIndicatorBG		Alpha		"0"		Accel 0.2 0.4

	RunEvent HudHealthBonusPulseLoop 0.4
}

// call to loop HudHealthBonusPulse
event HudHealthBonusPulseLoop
{
	RunEvent HudHealthBonusPulse 0.0
}

event HudHealthBonusPulseStop
{
	StopEvent HudHealthBonusPulse 0.0
	StopEvent HudHealthBonusPulseLoop 0.0

	Animate	HealthIndicatorBG		BgColor		"sh_blank"		Accel 0.0 0.0
	Animate	HealthIndicatorBG		Alpha		"0"		Accel 0.0 0.0
}

//Health Dying Pulse
event HudHealthDyingPulse
{
	Animate	PlayerStatusHealthBonusImage	Alpha	"0"		Linear 0.0 0.0

	Animate	HealthIndicatorBG		BgColor		"LowHealth"		Accel 0.0 0.0
	Animate	HealthIndicatorBG		Alpha		"200"		Accel 0.0 0.2
	Animate	HealthIndicatorBG		Alpha		"0"		Accel 0.2 0.4

	RunEvent HudHealthDyingPulseLoop 0.4
}

// call to loop HudHealthDyingPulse
event HudHealthDyingPulseLoop
{
	RunEvent HudHealthDyingPulse 0.0
}

event HudHealthDyingPulseStop
{
	StopEvent HudHealthDyingPulse 0.0
	StopEvent HudHealthDyingPulseLoop 0.0

	Animate	HealthIndicatorBG		BgColor		"Blank"		Accel 0.0 0.0
	Animate	HealthIndicatorBG		Alpha		"0"		Accel 0.0 0.0
}
